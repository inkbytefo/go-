Sınıf Tanımlama

package main

import "fmt"

class Person {
    private:
        string name
        int age
    
    public:
        Person(string name, int age) {
            this.name = name
            this.age = age
        }
        
        string getName() {
            return this.name
        }
        
        int getAge() {
            return this.age
        }
        
        void birthday() {
            this.age++
        }
}

func main() {
    person := Person("Ahmet", 30)
    fmt.Println("İsim:", person.getName())
    fmt.Println("Yaş:", person.getAge())
    
    person.birthday()
    fmt.Println("Yeni yaş:", person.getAge())
}

Şablon Kullanımı

package main

import "fmt"

template <typename T>
class Stack {
    private:
        []T items
    
    public:
        void push(T item) {
            this.items = append(this.items, item)
        }
        
        T pop() {
            if len(this.items) == 0 {
                panic("Stack boş")
            }
            
            n := len(this.items) - 1
            item := this.items[n]
            this.items = this.items[:n]
            return item
        }
        
        bool isEmpty() {
            return len(this.items) == 0
        }
}

func main() {
    intStack := Stack<int>{}
    intStack.push(1)
    intStack.push(2)
    intStack.push(3)
    
    fmt.Println(intStack.pop())  // 3
    fmt.Println(intStack.pop())  // 2
    fmt.Println(intStack.pop())  // 1
    
    stringStack := Stack<string>{}
    stringStack.push("Merhaba")
    stringStack.push("GO+")
    
    fmt.Println(stringStack.pop())  // GO+
    fmt.Println(stringStack.pop())  // Merhaba
}

İstisna İşleme

package main

import "fmt"

func divide(a, b int) int {
    if b == 0 {
        throw "Sıfıra bölme hatası"
    }
    return a / b
}

func main() {
    try {
        result := divide(10, 2)
        fmt.Println("Sonuç:", result)
        
        result = divide(10, 0)
        fmt.Println("Bu satır çalışmayacak")
    } catch (e) {
        fmt.Println("Hata yakalandı:", e)
    } finally {
        fmt.Println("İşlem tamamlandı")
    }
}
